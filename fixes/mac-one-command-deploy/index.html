// mac-xen one-command deploy · 2026-05-23

One paste.
All cascade-close.

Bundles every remaining mac-side execution into a single ssh-paste-and-go script. Apply EX_CONFIG fix, deploy worker, add omnimind route, snapshot xvm — all in sequence with verify gates between each step. Idempotent where possible. Bails fast on failure with named-step diagnosis.
Prerequisites assumed already in place on m4:
  • wrangler installed + logged in (brew install cloudflare-wrangler && wrangler login — see /ios-shortcut-deploy/)
  • CF account access for xlrd.org zone (for tools.xlrd.org route)
  • ~/.xen/secrets.env exists with XEN_MAC_TOKEN set
#!/bin/bash
# mac-xen · one-command deploy · cascade-close all remaining threads
# Save as ~/bin/xen-cascade-deploy.sh, chmod +x, then run.
set -e
set -o pipefail

LOG=~/.xen/state/cascade-deploy.log
mkdir -p ~/.xen/state ~/.xen/firehose-scripts ~/.xen/firehose-logs
exec > >(tee -a "$LOG") 2>&1
echo "=== cascade-deploy start $(date -u +%FT%TZ) ==="

# ── Step 1: EX_CONFIG fix (Fix B — local-disk relocation, more permanent) ──
echo "[1/5] EX_CONFIG fix · relocating Python scripts to ~/.xen/firehose-scripts/"
for s in beeper_firehose.py mail_firehose.py notif_firehose.py firehose_healthcheck.py beeper_live_index.py; do
  if [ -f "/Volumes/tech_/qi_data/exedus/xen/$s" ]; then
    cp -n "/Volumes/tech_/qi_data/exedus/xen/$s" ~/.xen/firehose-scripts/ || true
    echo "  + copied $s"
  fi
done

# Patch each plist if it exists
for plist in \
  ~/Library/LaunchAgents/com.xen.beeper-firehose.plist \
  ~/Library/LaunchAgents/com.xen.mail-firehose.plist \
  ~/Library/LaunchAgents/com.exedus.xen-notif-firehose.plist \
  ~/Library/LaunchAgents/com.xen.firehose-healthcheck.plist \
  ~/Library/LaunchAgents/com.xen.beeper-live-index.plist \
  ~/Library/LaunchAgents/com.xen.omnimind-heal.plist
do
  [ -f "$plist" ] || continue
  LABEL=$(basename "$plist" .plist)
  SCRIPT_NAME=$(echo "$LABEL" | sed 's/^com\.[^.]*\.//; s/-/_/g').py
  if [ -f "$HOME/.xen/firehose-scripts/$SCRIPT_NAME" ]; then
    plutil -replace ProgramArguments.1 -string "$HOME/.xen/firehose-scripts/$SCRIPT_NAME" "$plist"
    plutil -replace WorkingDirectory -string "$HOME/.xen/firehose-scripts" "$plist"
    plutil -replace StandardOutPath -string "$HOME/.xen/firehose-logs/$LABEL.out.log" "$plist"
    plutil -replace StandardErrorPath -string "$HOME/.xen/firehose-logs/$LABEL.err.log" "$plist"
    launchctl bootout gui/$(id -u)/$LABEL 2>/dev/null || true
    launchctl bootstrap gui/$(id -u) "$plist"
    echo "  ✓ $LABEL patched + reloaded"
  fi
done

# ── Step 2: Verify EX_CONFIG fix ──
echo "[2/5] verify launchd state"
launchctl list | grep -iE "firehose|beeper-live|omnimind-heal" | head -10

# ── Step 3: Deploy tools.xlrd.org worker ──
echo "[3/5] tools.xlrd.org worker deploy"
if [ ! -d ~/tools-runtime ]; then
  mkdir -p ~/tools-runtime/src
  curl -fsSL https://hitthe.link/tools-worker/worker.js -o ~/tools-runtime/src/index.js
  cat > ~/tools-runtime/wrangler.toml <<'EOF'
name = "tools-runtime"
main = "src/index.js"
compatibility_date = "2026-05-23"

[[routes]]
pattern = "tools.xlrd.org/*"
zone_name = "xlrd.org"
EOF
  echo "  ! seeded ~/tools-runtime · NOW set secrets + KV namespace:"
  echo "      cd ~/tools-runtime"
  echo "      wrangler secret put XEN_TOOLS_TOKEN  # paste a strong random token"
  echo "      wrangler secret put XEN_HMAC_KEY     # paste another strong random key"
  echo "      wrangler secret put XEN_MAC_TOKEN    # paste \$XEN_MAC_TOKEN value from ~/.xen/secrets.env"
  echo "      wrangler kv namespace create TOOLS_RATE"
  echo "      # add the returned id to wrangler.toml under [[kv_namespaces]]"
  echo "      wrangler deploy"
  echo "  ! then rerun this script to skip seeding + verify deployment"
else
  cd ~/tools-runtime
  if [ -f wrangler.toml ] && grep -q kv_namespaces wrangler.toml; then
    wrangler deploy 2>&1 | tail -5
    HEALTH=$(curl -s --max-time 5 https://tools.xlrd.org/health)
    echo "  health: $HEALTH"
  else
    echo "  ! KV namespace not bound in wrangler.toml · run kv namespace create + paste id first"
  fi
fi

# ── Step 4: omnimind /api/dictate-inject route ──
echo "[4/5] omnimind /api/dictate-inject route"
OMNIMIND_JS=$(find ~ /Volumes/tech_ -name "omnimind.js" -not -path "*/node_modules/*" 2>/dev/null | head -1)
if [ -f "$OMNIMIND_JS" ]; then
  if ! grep -q "/api/dictate-inject" "$OMNIMIND_JS"; then
    cp "$OMNIMIND_JS" "$OMNIMIND_JS.bak.cascade-deploy"
    # Append route handler — naive but works for express-like patterns:
    cat >> "$OMNIMIND_JS" <<'JS'

// Added by xen-cascade-deploy — /dictate web-path receiver
app.post('/api/dictate-inject', express.json(), (req, res) => {
  const { source, ts, text } = req.body || {};
  if (!text) return res.status(400).json({ error: 'no text' });
  const tag = source === 'xen-dictate' ? '[DICTATE]' : '[DICTATE/?]';
  injectIntoPane(`${tag} ${text}`);
  res.json({ ok: true, ts: ts || Date.now() });
});
JS
    echo "  ✓ route appended to $OMNIMIND_JS · restart omnimind to activate"
    echo "    launchctl bootout gui/$(id -u)/com.xen.omnimind-4441"
    echo "    launchctl bootstrap gui/$(id -u) ~/Library/LaunchAgents/com.xen.omnimind-4441.plist"
  else
    echo "  ✓ route already present"
  fi
else
  echo "  ! omnimind.js not found · manual route addition needed"
fi

# ── Step 5: UTM xvm snapshot ──
echo "[5/5] UTM xvm snapshot"
XVM_UUID="E2BD1556-24D0-42DC-8A09-D37A7B3EE9C1"
SNAP_NAME="xvm-cascade-deploy-$(date +%Y%m%d-%H%M%S)"
osascript <<EOF || echo "  ! AppleScript snapshot path failed · use UTM GUI: right-click xvm -> Snapshot -> Save"
tell application "UTM"
  set vmList to virtual machines
  repeat with v in vmList
    if id of v is "$XVM_UUID" then
      try
        save v as snapshot named "$SNAP_NAME"
        return "snapshot: $SNAP_NAME"
      end try
    end if
  end repeat
end tell
EOF

echo "=== cascade-deploy complete $(date -u +%FT%TZ) ==="
echo "Verify cascade:"
echo "  curl https://tools.xlrd.org/health"
echo "  curl -X POST https://xen.xlrd.org/api/dictate-inject -H 'Content-Type: application/json' -d '{\"source\":\"test\",\"text\":\"hello\"}'"
echo "  launchctl list | grep firehose"
echo "  curl -s https://xen.xlrd.org/events | head -2  # should be live SSE, not stale"
